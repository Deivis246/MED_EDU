let estudiante = {};
let resultados = JSON.parse(localStorage.getItem("resultadosMedInterna")) || {};

      {q:"¿Qué órgano puede afectarse por hipertensión crónica?", a:["Riñón","Cabello","Uñas"], c:0},
      {q:"Una medida no farmacológica importante es:", a:["Reducir sal","Aumentar tabaco","Evitar controles"], c:0}
    ]
  }
];

function ingresar(){
  let nombre=document.getElementById("nombre").value.trim();
  let cedula=document.getElementById("cedula").value.trim();
  let curso=document.getElementById("curso").value;

  if(nombre==="" || cedula==="" || curso===""){
    alert("Complete nombre, cédula y curso.");
    return;
  }

  estudiante={nombre,cedula,curso};
  localStorage.setItem("estudianteMedInterna",JSON.stringify(estudiante));
  iniciarApp();
}

function iniciarApp(){
  estudiante=JSON.parse(localStorage.getItem("estudianteMedInterna"));
  document.getElementById("login").style.display="none";
  document.getElementById("app").style.display="block";
  document.getElementById("userData").innerHTML=
    "<b>"+estudiante.nombre+"</b><br>C.I. "+estudiante.cedula+"<br>"+estudiante.curso;
  cargarModulos();
}

function cargarModulos(){
  let cont=document.getElementById("modules");
  cont.innerHTML="";
  modulos.forEach(m=>{
    let r=resultados[m.id] || {avance:0};
    cont.innerHTML+=`
      <div class="module">
        <div class="banner"></div>
        <div class="module-content">
          <h2>${m.titulo}</h2>
          <p>${m.desc}</p>
          <div class="progress"><div class="bar" style="width:${r.avance}%"></div></div>
          <p>${r.avance}% completado</p>
          <button onclick="abrirModulo('${m.id}')">Ingresar</button>
        </div>
      </div>
    `;
  });
}

function abrirModulo(id){
  let m=modulos.find(x=>x.id===id);
  document.getElementById("dashboard").style.display="none";
  document.getElementById("progreso").style.display="none";

  let html=`
    <h2>${m.titulo}</h2>
    <span class="badge">Estudiante: ${estudiante.nombre}</span>

    <h3>1. Pretest</h3>
    ${generarPreguntas(m,"pre")}

    <h3>2. Información del módulo</h3>
    <p>${m.info}</p>

    <h3>3. Caso clínico</h3>
    <div class="question">
      <p>${m.caso}</p>
      <input type="text" id="caso_${m.id}" placeholder="Escriba su respuesta">
    </div>

    <h3>4. Juego / cuestionario rápido</h3>
    <p>Responda nuevamente para reforzar el aprendizaje.</p>
    ${generarPreguntas(m,"juego")}

    <h3>5. Postest</h3>
    ${generarPreguntas(m,"post")}

    <button onclick="calificarModulo('${m.id}')">Finalizar módulo</button>
    <button onclick="mostrarDashboard()">Volver</button>

    <div id="resultado_${m.id}" class="result"></div>
  `;

  let sec=document.getElementById("contenidoModulo");
  sec.innerHTML=html;
  sec.style.display="block";
}

function generarPreguntas(m,tipo){
  let html="";
  m.preguntas.forEach((p,i)=>{
    html+=`<div class="question"><b>${i+1}. ${p.q}</b>`;
    p.a.forEach((op,j)=>{
      html+=`
        <label class="option">
          <input type="radio" name="${tipo}_${m.id}_${i}" value="${j}"> ${op}
        </label>`;
    });
    html+=`</div>`;
  });
  return html;
}

function calcular(m,tipo){
  let score=0;
  m.preguntas.forEach((p,i)=>{
    let r=document.querySelector(`input[name="${tipo}_${m.id}_${i}"]:checked`);
    if(r && parseInt(r.value)===p.c) score++;
  });
  return Math.round((score/m.preguntas.length)*100);
}

function calificarModulo(id){
  let m=modulos.find(x=>x.id===id);

  let pre=calcular(m,"pre");
  let juego=calcular(m,"juego");
  let post=calcular(m,"post");

  let casoResp=document.getElementById("caso_"+id).value.toLowerCase();
  let caso=casoResp.includes("hipotiroidismo") || casoResp.includes("diabetes") || casoResp.includes("hipertensión") || casoResp.includes("hipertension") ? 100 : 50;

  let final=Math.round((pre*0.2)+(caso*0.2)+(juego*0.2)+(post*0.4));

  resultados[id]={
    pretest:pre,
    caso:caso,
    juego:juego,
    postest:post,
    final:final,
    avance:100
  };

  localStorage.setItem("resultadosMedInterna",JSON.stringify(resultados));

  document.getElementById("resultado_"+id).innerHTML=`
    <h3>Resultado final</h3>
    <p>Pretest: ${pre}/100</p>
    <p>Caso clínico: ${caso}/100</p>
    <p>Juego: ${juego}/100</p>
    <p>Postest: ${post}/100</p>
    <h2>Nota final: ${final}/100</h2>
  `;

  cargarModulos();
}

function mostrarDashboard(){
  document.getElementById("dashboard").style.display="block";
  document.getElementById("contenidoModulo").style.display="none";
  document.getElementById("progreso").style.display="none";
  cargarModulos();
}

function mostrarProgreso(){
  document.getElementById("dashboard").style.display="none";
  document.getElementById("contenidoModulo").style.display="none";
  document.getElementById("progreso").style.display="block";

  let html="<table border='1' width='100%' cellpadding='8'><tr><th>Módulo</th><th>Pretest</th><th>Caso</th><th>Juego</th><th>Postest</th><th>Final</th></tr>";

  modulos.forEach(m=>{
    let r=resultados[m.id];
    if(r){
      html+=`<tr>
        <td>${m.titulo}</td>
        <td>${r.pretest}</td>
        <td>${r.caso}</td>
        <td>${r.juego}</td>
        <td>${r.postest}</td>
        <td>${r.final}</td>
      </tr>`;
    }
  });

  html+="</table>";
  document.getElementById("tablaResultados").innerHTML=html;
  document.getElementById("progresoTexto").innerHTML=
    "Estudiante: <b>"+estudiante.nombre+"</b> | Cédula: "+estudiante.cedula;
}

function cerrarSesion(){
  localStorage.removeItem("estudianteMedInterna");
  location.reload();
}

if(localStorage.getItem("estudianteMedInterna")){
  iniciarApp();
}
